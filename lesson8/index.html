Не выполняя кода, ответить, что выведет браузер и почему: if (!("a" in window)) { var a = 1; } alert(a); Браузер выведет undefined, потому что по условию if строка должна быть неистинной, без значения, а она имеет числовое значение 1, значит условие не соблюдено, а на этот случай условия вывода нет, т.е. в alert будет выведено undefined. var b = function a(x) { x && a(--x); }; alert(a); Браузер не выведет ничего, так как "а" - это имя функции, а в теле функции не определено какое-либо действие. function a(x) { return x * 2; } var a; alert(a); Браузер не выведет ничего, т.к. переменная "a" была объявлена без указания значения, а значит в "alert" выводится функция "а", для которой значение переменной "x", указанной в функции, отсутствует. function b(x, y, a) { arguments[2] = 10; alert(a); } b(1, 2, 3); Браузер выведет в "alert" значение третьего аргумента, которому присвоено значение 10 в теле функции. function a() { alert(this); } a.call(null); "this" обозначает объект, который запускает функцию. здесь объект отсутствует, значит "this" ничего не обозначает. Т.е. должен быть объект, но браузер не видит какой. Поэтому в "alert" будет указано, что это объект без названия.
